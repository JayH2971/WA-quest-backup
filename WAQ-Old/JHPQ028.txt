Clock _failsafe_ 30.00:00
Clock _timer_ 1:00
Clock _delay_ 0:01 0:02

Item _reward_ random_recipe

--	Quest start-up:
    start timer _failsafe_

 _enter_ task:
    when pc enters village
    stop timer _failsafe_
    start timer _delay_
    start timer _timer_

_delay_ task:
    prompt 1010 yes _begin_ no _quit_

variable _begin_
_check1_ task:
   when skill Running is at least 50

_victory_ task:
    when _check1_ and _begin_
    give pc _reward_
    end quest

_fail_ task:
    when _begin_ and not _check1_
    say 1003
    end quest

_timer_ task:
   end quest

_failsafe_ task:
   end quest

_quit_ task:
  say 1012
  end quest
